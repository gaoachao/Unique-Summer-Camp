## 递归实现

```javascript
var preorderTraversal = function(root) {
    const res=[];
    let preOrder = function(node,arr){
        if(node === null) return arr;
        arr.push(node.val);
        preOrder(node.left,arr);
        preOrder(node.right,arr);
    };
    preOrder(root,res);
    return res;
};
```
